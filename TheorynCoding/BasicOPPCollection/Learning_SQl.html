1. Table structures & example data

department

CREATE TABLE department (
    dept_id INT PRIMARY KEY AUTO_INCREMENT,
    dept_name VARCHAR(100) UNIQUE NOT NULL,
    location VARCHAR(100)
);

rank

CREATE TABLE rank (
    rank_id INT PRIMARY KEY AUTO_INCREMENT,
    rank_name VARCHAR(50),
    allowance_multiplier DECIMAL(5,2)
);

empl_details

CREATE TABLE empl_details (
    emp_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    design VARCHAR(50),
    rank_id INT,
    dept_id INT,
    FOREIGN KEY (rank_id) REFERENCES rank(rank_id),
    FOREIGN KEY (dept_id) REFERENCES department(dept_id)
);

salary

CREATE TABLE salary (
    salary_id INT PRIMARY KEY AUTO_INCREMENT,
    emp_id INT NOT NULL,
    basic DECIMAL(10,2),
    hra DECIMAL(10,2),
    allowance DECIMAL(10,2),
    total DECIMAL(10,2) GENERATED ALWAYS AS (basic + hra + allowance) STORED,
    FOREIGN KEY (emp_id) REFERENCES empl_details(emp_id)
);

2. DDL, DML, DCL, TCL — quick reference

DDL (Data Definition)

  • CREATE, ALTER, DROP, TRUNCATE
-- Add bonus column
ALTER TABLE salary ADD COLUMN bonus DECIMAL(10,2) DEFAULT 0;

DML (Data Manipulation)

  • INSERT, UPDATE, DELETE, SELECT
-- Update design & rank
UPDATE empl_details
SET design = 'Senior Engineer', rank_id = 2
WHERE emp_id = 1;

DCL (Permissions)

GRANT SELECT, INSERT ON salary TO 'report_user'@'localhost';
REVOKE INSERT ON salary FROM 'report_user'@'localhost';

TCL (Transactions)

START TRANSACTION;
UPDATE salary SET basic = 60000 WHERE emp_id = 1;
SAVEPOINT before_bonus;
UPDATE salary SET bonus = 5000 WHERE emp_id = 1;
ROLLBACK TO before_bonus;
COMMIT;

3. Advanced features: Triggers, Functions, Stored Procedures

Trigger — insert salary after new employee

CREATE TRIGGER after_emp_insert
AFTER INSERT ON empl_details
FOR EACH ROW
BEGIN
    INSERT INTO salary (emp_id, basic, hra, allowance) VALUES (NEW.emp_id, 30000, 5000, 2000);
END;

Function — annual salary

CREATE FUNCTION annual_salary(emp INT) RETURNS DECIMAL(12,2)
BEGIN
    DECLARE sal DECIMAL(12,2);
    SELECT total INTO sal FROM salary WHERE emp_id = emp;
    RETURN sal * 12;
END;

Stored Procedure — update salary by percent

CREATE PROCEDURE update_salary(IN emp INT, IN percent DECIMAL(5,2))
BEGIN
    UPDATE salary
    SET basic = basic + (basic * percent / 100),
        hra = hra + (hra * percent / 100),
        allowance = allowance + (allowance * percent / 100)
    WHERE emp_id = emp;
END;

4. Common & advanced queries (examples)

Find all employees in IT

SELECT e.name, d.dept_name
FROM empl_details e
JOIN department d ON e.dept_id = d.dept_id
WHERE d.dept_name = 'IT'
ORDER BY e.name;

Total salary per employee

SELECT e.name, (s.basic + s.hra + s.allowance) AS total_salary
FROM empl_details e
JOIN salary s ON e.emp_id = s.emp_id;

Employees earning more than department average

SELECT e.name, d.dept_name, (s.basic + s.hra + s.allowance) AS total_salary
FROM empl_details e
JOIN salary s ON e.emp_id = s.emp_id
JOIN department d ON e.dept_id = d.dept_id
WHERE (s.basic + s.hra + s.allowance) >
  (SELECT AVG(s2.basic + s2.hra + s2.allowance)
   FROM salary s2
   JOIN empl_details e2 ON s2.emp_id = e2.emp_id
   WHERE e2.dept_id = d.dept_id);

Top 3 earners

SELECT e.name, (s.basic + s.hra + s.allowance) AS total_salary
FROM salary s
JOIN empl_details e ON s.emp_id = e.emp_id
ORDER BY total_salary DESC
LIMIT 3;

Handle NULL salary parts

SELECT e.name, COALESCE(s.basic,0) + COALESCE(s.hra,0) + COALESCE(s.allowance,0) AS total_salary
FROM empl_details e
LEFT JOIN salary s ON e.emp_id = s.emp_id;

5. Tricky SQL gotchas & logical execution order

Logical execution order (important for tricky questions):

FROM -> WHERE -> GROUP BY -> HAVING -> SELECT -> DISTINCT -> ORDER BY -> LIMIT/OFFSET

HAVING vs ORDER BY

You must use HAVING after GROUP BY and before ORDER BY. Placing HAVING after ORDER BY is a syntax error.

Using aggregate functions

-- WRONG: WHERE SUM(s.basic) > 50000
-- RIGHT: use HAVING for aggregates
GROUP BY d.dept_name
HAVING SUM(s.basic) > 50000;

ORDER BY on non-grouped column

Some SQL engines are strict — ordering by a column not in GROUP BY (and not aggregated) can error.

6. Recursive CTEs & Window Functions

Recursive query — reporting chain

WITH RECURSIVE reporting_chain(emp_id, name, manager_id, depth) AS (
  SELECT emp_id, name, manager_id, 1
  FROM employee_hierarchy
  WHERE manager_id = 2
  UNION ALL
  SELECT e.emp_id, e.name, e.manager_id, rc.depth + 1
  FROM employee_hierarchy e
  JOIN reporting_chain rc ON e.manager_id = rc.emp_id
  WHERE rc.depth <= 5
)
SELECT * FROM reporting_chain;

Window function examples

-- cumulative salary
SELECT e.name, (s.basic + s.hra + s.allowance) AS total_salary,
       SUM(s.basic + s.hra + s.allowance) OVER (ORDER BY e.emp_id) AS cumulative_salary
FROM empl_details e
JOIN salary s ON e.emp_id = s.emp_id;

-- rank within dept
SELECT e.name, d.dept_name,
       (s.basic + s.hra + s.allowance) AS total_salary,
       RANK() OVER (PARTITION BY d.dept_id ORDER BY (s.basic + s.hra + s.allowance) DESC) AS rank_in_dept
FROM empl_details e
JOIN salary s ON e.emp_id = s.emp_id
JOIN department d ON e.dept_id = d.dept_id;

7. Indexes, EXPLAIN & performance tuning

Index examples

Using EXPLAIN

EXPLAIN SELECT e.name FROM empl_details e
JOIN salary s ON e.emp_id = s.emp_id
WHERE s.basic > 50000;

Tips: index columns used in WHERE, JOIN and ORDER BY; avoid over-indexing; use EXPLAIN to find full table scans.
